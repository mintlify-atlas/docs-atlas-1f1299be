# Content Slider with Cards

Create dynamic content sliders that display multiple items at once, perfect for product showcases, feature highlights, or content feeds. This example demonstrates how to use custom React components with flexible scroll offsets and snapping behavior.

## Basic Content Slider

A simple content slider with card components:

```tsx
import { Carousel } from "react-any-carousel";

function ContentSlider() {
  const products = [
    { id: 1, name: "Wireless Headphones", price: "$99", rating: 4.5 },
    { id: 2, name: "Smart Watch", price: "$299", rating: 4.8 },
    { id: 3, name: "Laptop Stand", price: "$49", rating: 4.3 },
    { id: 4, name: "Mechanical Keyboard", price: "$149", rating: 4.7 },
    { id: 5, name: "USB-C Hub", price: "$79", rating: 4.6 },
  ];

  return (
    <Carousel
      scrollSnapType="start"
      scrollOffset={320}
      duration={600}
    >
      {products.map((product) => (
        <div
          key={product.id}
          style={{
            width: "280px",
            padding: "20px",
            margin: "0 20px 0 0",
            background: "white",
            borderRadius: "12px",
            boxShadow: "0 2px 8px rgba(0, 0, 0, 0.1)",
            flexShrink: 0,
          }}
        >
          <div
            style={{
              width: "100%",
              height: "180px",
              background: "linear-gradient(135deg, #667eea 0%, #764ba2 100%)",
              borderRadius: "8px",
              marginBottom: "16px",
            }}
          />
          <h3 style={{ margin: "0 0 8px 0", fontSize: "18px" }}>{product.name}</h3>
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
            <span style={{ fontSize: "20px", fontWeight: "bold", color: "#2563eb" }}>{product.price}</span>
            <span style={{ fontSize: "14px", color: "#fbbf24" }}>★ {product.rating}</span>
          </div>
        </div>
      ))}
    </Carousel>
  );
}
```

<Note>
  The `scrollOffset={320}` is set to scroll approximately one card at a time (280px card + margins).
</Note>

## Scroll Offset Strategies

Different strategies for controlling scroll behavior:

| Strategy | `scrollOffset` Value | Use Case |
|----------|---------------------|----------|
| **One at a time** | Card width + margin | Precise card-by-card navigation |
| **Multiple items** | (Card width × n) + margins | Scroll multiple cards at once |
| **Partial peek** | Less than card width | Show preview of next card |
| **Container width** | Container width | Page-style scrolling |

```tsx
// Example: Dynamic scroll offset based on container
import { Carousel } from "react-any-carousel";
import { useRef, useState, useEffect } from "react";

function DynamicOffsetSlider() {
  const containerRef = useRef<HTMLDivElement>(null);
  const [scrollOffset, setScrollOffset] = useState(400);

  useEffect(() => {
    if (containerRef.current) {
      // Set scroll offset to container width for "page" scrolling
      setScrollOffset(containerRef.current.offsetWidth);
    }
  }, []);

  const items = Array.from({ length: 12 }, (_, i) => ({ id: i + 1, title: `Item ${i + 1}` }));

  return (
    <div ref={containerRef} style={{ maxWidth: "1000px", margin: "0 auto" }}>
      <Carousel
        scrollSnapType="start"
        scrollOffset={scrollOffset}
        duration={600}
      >
        {items.map((item) => (
          <div
            key={item.id}
            style={{
              width: "300px",
              height: "180px",
              padding: "20px",
              margin: "0 15px 0 0",
              background: "linear-gradient(135deg, #667eea 0%, #764ba2 100%)",
              color: "white",
              borderRadius: "12px",
              display: "flex",
              alignItems: "center",
              justifyContent: "center",
              fontSize: "20px",
              fontWeight: "600",
              flexShrink: 0,
            }}
          >
            {item.title}
          </div>
        ))}
      </Carousel>
    </div>
  );
}
```

## Production Example: E-commerce Slider

A complete e-commerce product slider with all features:

```tsx
import { Carousel } from "react-any-carousel";
import { FiChevronRight, FiShoppingCart, FiHeart } from "react-icons/fi";
import { useState } from "react";

interface Product {
  id: number;
  name: string;
  price: number;
  originalPrice?: number;
  image: string;
  badge?: string;
  rating: number;
  reviews: number;
}

function ProductCard({ product }: { product: Product }) {
  const [liked, setLiked] = useState(false);
  const discount = product.originalPrice
    ? Math.round(((product.originalPrice - product.price) / product.originalPrice) * 100)
    : 0;

  return (
    <div
      style={{
        width: "280px",
        background: "white",
        borderRadius: "16px",
        overflow: "hidden",
        boxShadow: "0 2px 8px rgba(0, 0, 0, 0.08)",
        margin: "0 16px 0 0",
        flexShrink: 0,
        position: "relative",
        transition: "transform 0.2s, box-shadow 0.2s",
      }}
      onMouseEnter={(e) => {
        e.currentTarget.style.transform = "translateY(-4px)";
        e.currentTarget.style.boxShadow = "0 8px 24px rgba(0, 0, 0, 0.12)";
      }}
      onMouseLeave={(e) => {
        e.currentTarget.style.transform = "translateY(0)";
        e.currentTarget.style.boxShadow = "0 2px 8px rgba(0, 0, 0, 0.08)";
      }}
    >
      {product.badge && (
        <div
          style={{
            position: "absolute",
            top: "12px",
            left: "12px",
            background: "#ef4444",
            color: "white",
            padding: "4px 12px",
            borderRadius: "6px",
            fontSize: "12px",
            fontWeight: "600",
            zIndex: 1,
          }}
        >
          {product.badge}
        </div>
      )}
      <button
        onClick={() => setLiked(!liked)}
        style={{
          position: "absolute",
          top: "12px",
          right: "12px",
          background: "white",
          border: "none",
          borderRadius: "50%",
          width: "40px",
          height: "40px",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          cursor: "pointer",
          zIndex: 1,
          boxShadow: "0 2px 8px rgba(0, 0, 0, 0.1)",
        }}
      >
        <FiHeart size={20} fill={liked ? "#ef4444" : "none"} color={liked ? "#ef4444" : "#6b7280"} />
      </button>
      <div
        style={{
          width: "100%",
          height: "240px",
          background: `url(${product.image})`,
          backgroundSize: "cover",
          backgroundPosition: "center",
        }}
      />
      <div style={{ padding: "20px" }}>
        <h3 style={{ margin: "0 0 8px 0", fontSize: "16px", fontWeight: "600", color: "#111827" }}>{product.name}</h3>
        <div style={{ display: "flex", alignItems: "center", gap: "6px", marginBottom: "12px" }}>
          <span style={{ color: "#fbbf24", fontSize: "14px" }}>★</span>
          <span style={{ fontSize: "14px", color: "#6b7280" }}>
            {product.rating} ({product.reviews})
          </span>
        </div>
        <div style={{ display: "flex", alignItems: "center", gap: "8px", marginBottom: "16px" }}>
          <span style={{ fontSize: "24px", fontWeight: "700", color: "#111827" }}>${product.price}</span>
          {product.originalPrice && (
            <>
              <span style={{ fontSize: "16px", color: "#9ca3af", textDecoration: "line-through" }}>
                ${product.originalPrice}
              </span>
              <span style={{ fontSize: "14px", color: "#ef4444", fontWeight: "600" }}>-{discount}%</span>
            </>
          )}
        </div>
        <button
          style={{
            width: "100%",
            padding: "12px",
            background: "#2563eb",
            color: "white",
            border: "none",
            borderRadius: "8px",
            fontSize: "14px",
            fontWeight: "600",
            cursor: "pointer",
            display: "flex",
            alignItems: "center",
            justifyContent: "center",
            gap: "8px",
          }}
        >
          <FiShoppingCart size={18} />
          Add to Cart
        </button>
      </div>
    </div>
  );
}
```

## Key Takeaways

- **Flexible Content**: Use any React component as carousel children
- **Multiple Visible**: Control how many items appear simultaneously with `scrollOffset`
- **Snapping Behavior**: Choose `start`, `center`, `end`, or `none` based on your layout
- **Interactive Elements**: Buttons and links inside carousel items work seamlessly
- **Responsive Design**: Adjust card sizes and scroll offsets based on viewport
